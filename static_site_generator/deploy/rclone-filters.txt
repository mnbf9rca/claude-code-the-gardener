# rclone filter rules for R2 sync
# IMPORTANT: rclone processes filters in order, FIRST MATCH WINS
# Includes (+) must appear BEFORE excludes (-) for exceptions to work

# Include .claude FIRST (exception to hidden file rule)
+ .claude/
+ .claude/**

# Now exclude all other hidden files/directories
- .*
- .*/**

# Exclude git directories (belt and suspenders)
- .git/
- .git/**

# Include everything else
+ *
+ **
